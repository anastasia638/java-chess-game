package modele.jeu;

import modele.plateau.Case;
import modele.plateau.Plateau;

public class Jeu {
    private Plateau plateau;
    public Coup coupRecu; // rendre coupRecu public pour l'accès dans Joueur

    public Jeu() {
        plateau = new Plateau();
        plateau.placerPieces();
    }

    public Plateau getPlateau() {
        return plateau;
    }

    public void envoyerCoup(Coup c) {
        plateau.deplacerPiece(c.dep, c.arr);
        plateau.notifyObservers(); // Notifie la vue
        System.out.println("Coup joué de (" + c.dep.getX() + "," + c.dep.getY() + ") → (" + c.arr.getX() + "," + c.arr.getY() + ")");
        coupRecu = c; // Assigner le coup reçu
    }
}
