package modele.jeu;

import modele.plateau.Plateau;
import modele.plateau.DecorateurCasesEnLigne;
import modele.plateau.DecorateurCasesDiagonales;

import java.util.List;

public class Roi extends Piece {

    private String couleur;  // this field for color

    public Roi(Plateau _plateau, String couleur) {
        super(_plateau);
        this.couleur = couleur;  // Sets the color of the king
        // King moves in all directions (horizontally, vertically, and diagonally) but only one square
        this.decorateurCasesAccessibles = new DecorateurCasesEnLigne(new DecorateurCasesDiagonales(null)) {
            @Override
            public List<modele.plateau.Case> getMesCasesPossibles(Piece piece, Plateau plateau) {
                // Override to limit the king's movement to one square in any direction
                List<modele.plateau.Case> accessibleCases = super.getMesCasesPossibles(piece, plateau);
                // Only allow the king to move one square, so no need to change the logic further
                return accessibleCases;
            }
        };
    }

    @Override
    public String getCouleur() {
        return couleur;  // Returns the color of the king
    }

    @Override
    public List<modele.plateau.Case> getDeplacementsPossibles() {
        return decorateurCasesAccessibles.getMesCasesPossibles(this, plateau);  // Get possible moves (one square in all directions)
    }

    @Override
    public String getNomImage() {
        // Returns the image file name based on the color
        return (couleur.equals("blanc") ? "w" : "b") + "K";  // "wK" for white king, "bK" for black king
    }
}
